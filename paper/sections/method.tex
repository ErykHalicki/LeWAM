\section{Method}

\subsection{Overview}

LeWAM consists of three components: (1) a \textbf{video encoder} that maps video frames to per-patch latent embeddings; (2) a \textbf{flow-matching DiT} that predicts future latent embeddings and actions conditioned on past frames; and (3) a VLM for visually aware language conditioning. 
Figure~\ref{fig:arch} illustrates the full system.

\subsection{Video Encoder}

We use V-JEPA~2.1~\citep{vjepa2_1} ViT-B as our visual encoder. The encoder is initialized from the V-JEPA~2.1 pretrained weights and is frozen during training. V-JEPA2 has shown very strong tranfer learning properties.

%---------------------------------------

\subsection{Vision Language Encoder}

We use SmolVLM2-256M for its compact size and for a consist reference point with SmolVLA. During training, the language conditioning is dropped with probability $p_{\text{drop}} = 0.15$ to enable classifier-free guidance at inference time \cite{DreamZero}. \subsection{Latent Predictor: Flow-Matching DiT}

Given the patch embeddings for $P$ past frames $z_{\text{past}} \in \mathbb{R}^{P H_p W_p \times D_{\text{enc}}}$, the DiT predicts the patch embeddings of $F$ future frames $z_{\text{fut}} \in \mathbb{R}^{F H_p W_p \times D_{\text{enc}}}$ and $A$ future actions.


\paragraph{Architecture.}
\begin{enumerate}
    \item \textbf{Self-attention} over noisy future tokens with 3D-RoPE positional encoding and adaLN-Zero conditioning on the timestep embedding $\tau(t)$.
    \item \textbf{Cross-attention} to past frame tokens (with 3D-RoPE), language tokens (no RoPE), and a state token (no RoPE), with all sources concatenated as keys and values.
    \item \textbf{MLP} with adaLN-Zero conditioning.
\end{enumerate}

At inference, $z_{\text{fut}}$ is recovered by integrating the learned velocity field from $t=0$ to $t=1$ using an ODE solver.

%---------------------------------------

\subsection{Inverse Dynamics Model}

The IDM predicts an action chunk $\hat{a}_{1:K} \in \mathbb{R}^{K \times D_a}$ from the current and future patch embeddings.

\paragraph{Architecture.}
The IDM maintains $K$ learned action query tokens and applies $N_{\text{IDM}} = 6$ transformer blocks, each consisting of:
\begin{enumerate}
    \item \textbf{Cross-attention} from action queries to current frame tokens and future frame tokens (both with 3D-RoPE), and to a state token.
    \item \textbf{Self-attention} over action queries (bidirectional, no positional encoding).
    \item \textbf{MLP}.
\end{enumerate}
The IDM uses standard LayerNorm throughout (no adaLN). The final action token representations are projected to $D_a$ via a linear head.

\paragraph{Training.}
The IDM is supervised with an MSE loss on the action chunk:
\begin{equation}
    \mathcal{L}_{\text{IDM}} = \left\| \hat{a}_{1:K} - a_{1:K}^* \right\|^2.
\end{equation}
To bridge the train/test gap between ground-truth and predicted future embeddings, we use \textbf{teacher forcing}: with probability $p_{\text{tf}}$, the IDM is conditioned on the ground-truth future patches; otherwise on embeddings generated by the DiT via ODE solving. \subsection{Training Objectives}

The combined training loss is:
\begin{equation}
    \mathcal{L} = \mathcal{L}_{\text{pred}} + \lambda_{\text{IDM}}\, \mathcal{L}_{\text{IDM}} + \lambda_{\text{SIGReg}}\, \mathcal{L}_{\text{SIGReg}}(Z),
\end{equation}
where $\mathcal{L}_{\text{SIGReg}}$ is the Sketched Isotropic Gaussian Regularization loss~\citep{LeJEPA} applied to the encoder output $Z$ to prevent representational collapse. During video-only pretraining (no action labels), the IDM term is dropped.

%---------------------------------------

\subsection{Data}

\paragraph{Video pretraining.}
Before finetuning on robot data, we pretrain the DiT and the V-JEPA~2 encoder jointly on Something-Something v2 (SSv2)~\citep{SomethingSomething}, a large-scale dataset of 220k crowd-sourced videos depicting humans performing 174 fine-grained manipulation categories (\emph{e.g.}\ ``pushing something from left to right'', ``pouring something into something'').
SSv2 is particularly well-suited for this purpose because its clips are action-centric, object-interaction-heavy, and temporally causal in a way that mirrors the structure of robot manipulation, unlike scene-centric datasets such as Kinetics.
Each clip is annotated with a natural-language action label, which we use as the language conditioning signal for the DiT, allowing the text and vision encoders to align on grounded, manipulation-relevant concepts before any robot data is seen.
During this stage the language encoder (T5Gemma-S) remains frozen and only the DiT and V-JEPA~2 encoder weights are updated.
The training objective is $\mathcal{L}_{\text{pred}} + \lambda_{\text{SIGReg}}\,\mathcal{L}_{\text{SIGReg}}$ (no IDM term, as SSv2 carries no action labels).

\subsection{Pseudocode}

\begin{algorithm}[h]
\caption{LeWAM Inference}
\begin{algorithmic}[1]
\Require video context $V_{\text{cur}}$, language instruction $\ell$, proprioceptive state $s$
\State $z_{\text{cur}} \gets \textsc{VJEPA2}(V_{\text{cur}})$ \Comment{per-patch embeddings}
\State $l \gets \textsc{T5Gemma}(\ell)$ \Comment{or $\mathbf{0}$ for unconditional}
\State $x_0 \sim \mathcal{N}(0, I)$
\State $z_{\text{fut}} \gets \textsc{ODESolve}(v_\theta,\, x_0,\, z_{\text{cur}},\, l,\, s)$ \Comment{integrate $t: 0 \to 1$}
\State $\hat{a}_{1:K} \gets \textsc{IDM}(z_{\text{cur}},\, z_{\text{fut}},\, s)$
\State \Return $z_{\text{fut}},\; \hat{a}_{1:K}$
\end{algorithmic}
\end{algorithm}

\begin{algorithm}[h]
\caption{LeWAM Training Step}
\begin{algorithmic}[1]
\Require video clip $(V_{\text{cur}}, V_{\text{fut}})$, instruction $\ell$, actions $a_{1:K}^*$, state $s$
\State $z_{\text{cur}} \gets \textsc{VJEPA2}(V_{\text{cur}})$;\quad $z_{\text{fut}} \gets \textsc{VJEPA2}(V_{\text{fut}})$
\State $l \gets \textsc{T5Gemma}(\ell)$ with prob.\ $1 - p_{\text{drop}}$, else $\mathbf{0}$ \Comment{CFG dropout}
\State Sample $t \sim \mathcal{U}[0,1]$,\; $x_0 \sim \mathcal{N}(0, I)$
\State $x_t \gets (1-t)\,x_0 + t\,z_{\text{fut}}$
\State $\mathcal{L}_{\text{pred}} \gets \|v_\theta(x_t, t, z_{\text{cur}}, l, s) - (z_{\text{fut}} - x_0)\|^2$
\State $\tilde{z}_{\text{fut}} \gets \textsc{ODESolve}(v_\theta,\, \mathcal{N}(0,I),\, z_{\text{cur}},\, l,\, s)$
\State $z_{\text{idm}} \gets z_{\text{fut}}$ with prob.\ $p_{\text{tf}}$, else $\tilde{z}_{\text{fut}}$ \Comment{teacher forcing}
\State $\mathcal{L}_{\text{IDM}} \gets \|\textsc{IDM}(z_{\text{cur}}, z_{\text{idm}}, s) - a_{1:K}^*\|^2$
\State $\mathcal{L} \gets \mathcal{L}_{\text{pred}} + \lambda_{\text{IDM}}\,\mathcal{L}_{\text{IDM}} + \lambda_{\text{SIGReg}}\,\mathcal{L}_{\text{SIGReg}}(z_{\text{cur}})$
\State Update $\theta$ via gradient descent on $\mathcal{L}$
\end{algorithmic}
\end{algorithm}
